Code Example

# Correlation & Covariance — EDA with pandas and seaborn
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

df = pd.read_csv("customer_churn.csv")
print(df.describe(include="all"))

# Missingness report
print((df.isna().mean() * 100).sort_values(ascending=False).head())

# Bivariate relationship
sns.boxplot(data=df, x="contract_type", y="monthly_charges")
plt.title("Correlation & Covariance: monthly charges by contract")
plt.tight_layout()
plt.savefig("eda_contract_charges.png")
