Code Example

# StatefulSets — Deployment + Service manifest
apiVersion: apps/v1
kind: Deployment
metadata:
  name: api-statefulsets
spec:
  replicas: 3
  selector:
    matchLabels:
      app: api-statefulsets
  template:
    metadata:
      labels:
        app: api-statefulsets
    spec:
      containers:
        - name: api
          image: devspark.azurecr.io/api:statefulsets
          ports:
            - containerPort: 8080
          readinessProbe:
            httpGet:
              path: /health
              port: 8080
            initialDelaySeconds: 5
          resources:
            requests:
              cpu: 100m
              memory: 128Mi
            limits:
              cpu: 500m
              memory: 512Mi
---
apiVersion: v1
kind: Service
metadata:
  name: api-statefulsets
spec:
  selector:
    app: api-statefulsets
  ports:
    - port: 80
      targetPort: 8080
  type: ClusterIP
